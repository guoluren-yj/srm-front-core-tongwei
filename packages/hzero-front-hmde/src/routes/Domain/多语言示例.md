
```js
import React from 'react';
import { Header, Content } from 'components/Page';
import { Button } from 'choerodon-ui/pro';
import intl from 'utils/intl';
import formatterCollections from 'utils/intl/formatterCollections';

const Domain = (props) => {

  const { history } = props;

  const handlePush = () => {
    history.push('/hmde/business-object/detail/1');
  };

  return (
    <>
      {/* 
        FIXME:
        所有中文描述都得写多语言，intl.get(key).d(title)
        key可以先随便写一个，后续统一调整，例如：hmde.xxx，
        title写对应的中文描述（按原型的title）
      */}
      <Header title={intl.get('hmde.xxx').d('领域下的业务对象')}>
        <Button onClick={handlePush}>跳转</Button>
      </Header>
      <Content>
        领域下的业务对象
      </Content>
    </>
  );
};

// FIXME: 多语言注解formatterCollections，只有加入了多语言才会生效
export default formatterCollections({ code: ['hmde.domain'] })(Domain);

```
